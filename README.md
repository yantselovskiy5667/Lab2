import java.io.PrintStream;
import java.util.Scanner;

public class Main {
    public static Scanner in = new Scanner(System.in);
    public static PrintStream out = System.out;

    public static void main(String[] args) {
        int n=in.nextInt();
        int x=in.nextInt();
        int k=0;
        int l=0;
        int p=-1;
        for (int i=0;i<n;i++){
            int a=in.nextInt();
            k+=a;
            l+= Math.abs(a);
            if(k>=x){
                p=i;
                break;
            }

        }
        if(p==-1) {
            out.println("OK");
            out.println(l);
        }
        else {
            out.println(p+1);
            out.println(l);
        }
    }
}


```

import java.io.PrintStream;
import java.util.Scanner;

public class Main {
    public static Scanner in = new Scanner(System.in);
    public static PrintStream out = System.out;

    public static void main(String[] args) {
        int n = in.nextInt();
        if (n <= 0) {
            out.print("ERROR");
        }
        else {
            int r1 = n % 10;
            while (n > 0) {
                int r2 = n % 10;
                if (r2 < r1)
                    r1 = r2;
                n = n / 10;
            }
            if (r1 < 0)
                out.print("ERROR");
            else
                out.print(r1);
        }
    }
}

```java

import java.io.PrintStream;
import java.util.Scanner;

public class Main {
    public static Scanner in = new Scanner(System.in);
    public static PrintStream out = System.out;

    public static void main(String[] args) {
        int n=in.nextInt();
        int iMax=0;
        int iMin=0;
        double []a=new double[n];
        for(int i=0;i<n;i++){
            a[i]=in.nextDouble();;
        }
        double max=a[0];
        double min=a[0];
        for(int i=1;i<n;i++) {
            if (a[i] > max) {
                max = a[i];
                iMax = i;
            }
            if (a[i] <min) {
                min = a[i];
                iMin = i;
            }
        }
        if(a[iMax]-a[iMin]<0.000001)
            for(int i=0;i<n;i++) {
                a[i]=0;
                out.print("1,000");
            }
        else
            for(int i=0;i<n;i++) {
            a[i]=(a[i]-a[iMin])/(a[iMax]-a[iMin]);
            out.printf("%.3f ",a[i]);
        }
        out.println();
    }
}




```
